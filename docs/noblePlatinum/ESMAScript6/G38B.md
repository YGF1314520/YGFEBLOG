# class静态成员

## 函数
```js
  function Phone() {}
  //静态成员
    Phone.name = "手机";
    Phone.change = function () {
      console.log("我可以改变世界");
      };
    Phone.prototype.size = "5.5inch";
    let nokia = new Phone();
    console.log(nokia.name);
    //   nokia.change();
    console.log(nokia.size);
```

## class

```js
    class Phone {
    // 静态属性
      static name = "手机";
      static change() {
        console.log("我可以改变世界");
      }
    }

    let nokis = new Phone();
    console.log(nokis.name);
    console.log(Phone.name);
```
